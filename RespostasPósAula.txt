def soma_lista(lista):
    soma = 0
    for numero in lista:
        soma += numero  
    return soma

#Quest達o 2
def maior_numero(lista):
    maior = lista[0]  
    for numero in lista:
        if numero > maior:
            maior = numero  
    return maior

def adiciona_palavra(palavras, nova_palavra):
    palavras += [nova_palavra]  
    return palavras

def numero_existe(lista, numero):
    for item in lista:
        if item == numero:  
            return True
    return False
